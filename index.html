JavaScript jQuery and AJAX

JavaScript is a client-side scripting language.

What is jQuery?

jQuery is a fast, small, and feature-rich JavaScript library. It makes things like HTML document traversal and manipulation, event handling, animation, and Ajax much simpler with an easy-to-use API that works across a multitude of browsers. With a combination of versatility and extensibility, jQuery has changed the way that millions of people write JavaScript.

AJAX Call using JSON

AJAX stand for Asynchronous JavaScript and XML. it is used for creating dynamic and fast web pages. Since AJAX is Asynchronous, it helps in updating a part of web page with out reloading the whole page.

JavaScript

$.ajax({
    url: "getPost.php",
    data: {
        postID: 5
    },
    success: function( data ) {
        $( "#post" ).html( "<strong>" + data.author + "</strong><br />" + data.content );
    }
});

<?php

/**
 * Car object - The functionality of the car allows it to be started and stopped, driven and brakes to be applied.  The car also records a record of the trip it has taken.
 */
Class Car {
    
    const ENGINE_ON = "on";
    const ENGINE_OFF = "off";
    
    private $model = null;
    private $engine_state = null;
    private $trip = null;
    
    public function __construct($model='Generic')
    {
        $this->model = $model;
        $this->engine_state = self::ENGINE_OFF;
        $this->record_trip($this->get_model() . " has been created.");
    }
    
    public function start_engine()
    {
        if ($this->is_engine_on()) {            
            throw new Exception("The car (" . $this->get_model() . ") engine is already on.", 1);
        } else {
            $this->engine_state = self::ENGINE_ON;
            $this->record_trip($this->get_model() . " engine has started.");
        }
    }
    
    public function stop_engine()
    {
        if ($this->is_engine_off()) {            
            throw new Exception("The car (" . $this->get_model() . ") engine is already off.", 1);
        } else {
            $this->engine_state = self::ENGINE_OFF;
            $this->record_trip($this->get_model() . " enging has stopped.");
        }
    }
    
    public function drive()
    {
        if ($this->is_engine_off()) {            
            throw new Exception("The car (" . $this->get_model() . ") engine is off and cannot be driven.", 1);
        } else {
            $this->record_trip($this->get_model() . " is driving.");
        }
    }
    
    public function brake()
    {
        if ($this->is_engine_off()) {            
            throw new Exception("The car (" . $this->get_model() . ") engine is off and brakes cannot be applied.", 1);
        } else {
            $this->record_trip($this->get_model() . " is braking.");
        }
    }
    
    public function record_trip($text = '')
    {
        $this->trip .= $text . "\n";
    }
    
    public function display_trip_record()
    {
        echo nl2br($this->trip);
    }
    
    public function get_model()
    {
        return $this->model;
    }
    
    public function is_engine_on()
    {
        return ($this->engine_state == self::ENGINE_ON);
    }
    
    public function is_engine_off()
    {
        return ($this->engine_state == self::ENGINE_OFF);
    }
}

/**
 * Ford object - The functionality of a ford car is that it allows the same functionality of a generic car with extra additions.  The extras include air conditioning.
 */
Class Ford extends Car {
    const MODEL_TYPE = "Ford";
    const AIRCON_ON = "on";
    const AIRCON_OFF = "off";
    
    private $aircon_state = null;
    
    public function __construct()
    {
        parent::__construct(self::MODEL_TYPE);
        $this->aircon_state = self::AIRCON_OFF;
    }
    
    public function start_aircon()
    {
        if ($this->is_engine_off()) {
            throw new Exception("The car (" . $this->get_model() . ") engine is off so air conditioning cannot be used.", 1);
        } else if ($this->is_aircon_on()) {            
            throw new Exception("The car (" . $this->get_model() . ") air conditioning is already on.", 1);
        } else {
            $this->aircon_state = self::AIRCON_ON;
            $this->record_trip($this->get_model() . " air conditioning has been turned " . $this->aircon_state . ".");
        }
    }
    
    public function stop_aircon()
    {
        if ($this->is_engine_off()) {            
            throw new Exception("The car (" . $this->get_model() . ") engine is switched off, therefore air conditioning is already off.", 1);
            $this->aircon_state = self::AIRCON_OFF;
        } else if ($this->is_aircon_off()) {            
            throw new Exception("The car (" . $this->get_model() . ") air conditioning is already off.", 1);
        } else {
            $this->aircon_state = self::AIRCON_OFF;
            $this->record_trip($this->get_model() . " air conditioning has been turned " . $this->aircon_state . ".");
        }
    }
    
    /**
     * stop_engine - Overloaded method to switch off additional features if they are currently turned on.
     */
    public function stop_engine()
    {
        if ($this->is_engine_off()) {            
            throw new Exception("The car (" . $this->get_model() . ") engine is already off.", 1);
        } else {
            if ($this->is_aircon_on()) {            
                $this->stop_aircon();
            } 
            $this->engine_state = self::ENGINE_OFF;
            $this->record_trip($this->get_model() . " enging has stopped.");
        }
    }
    
    public function is_aircon_on()
    {
        return ($this->aircon_state == self::AIRCON_ON);
    }
    
    public function is_aircon_off()
    {
        return ($this->aircon_state == self::AIRCON_OFF);
    }
}

try {
    $car = new Ford();
    $car->start_engine();
    $car->drive();
    $car->brake();
    $car->drive();
    $car->brake();
    $car->start_aircon();
    $car->drive();
    $car->drive();
    $car->stop_aircon();
    $car->stop_engine();

    $car->display_trip_record();
} catch (Exception $e) {
    die("EXCEPTION: " . $e->getMessage());
}
